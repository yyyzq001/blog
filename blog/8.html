    八、手写实现call(),apply()和bind()函数

    # 八、手写实现call(),apply()和bind()函数

    # 1.call()实现

    # 定义

    官方定义连接 (opens new window)

    语法

    Function.call(thisArg,arg1,arg2,...)

    参数

    thisArg: 可选的。function运行时的this值，但这个this可能会发生变化：在非严格模式下，该参数为null或undefined时会自动替换指向全局对象，原始值会被包装。

    arg1，arg2：

    传递给function的正常参数。

    官方效果：

    function A(name) {
    this.name = name
}

function B(name) {
    A.call(this, name)
}

let ss = new B("younima")

console.log(ss.name);//younima

    注：call函数对于function来说相当于借用，对B对象的操作并不会影响到A。

    # 手写实现

    /**
 * @description: 自定义call实现
 * @param thisArg   上下文this对象
 * @param ...arg    对应正常参数
 * @return {*}
 */

Function.prototype.newcall = function (thisArg, ...arg) {
    thisArg = typeof thisArg === "object" ? thisArg : window
    //保证不覆盖原有属性，实现与function原有属性不关联
    const key = Symbol()
    //this指的是对应的Function方法
    thisArg[key] = this
    //执行函数
    thisArg[key](...arg)
    //从对象清除该方法，保证对象的纯净
    delete thisArg[key]
}

//实验效果
function A(name) {
    this.name = name
}

function B(name) {
    A.newcall(this, name)
}

let ss = new B("younima")

console.log(ss.name);//younima

    # 2.apply()实现

    # 定义

    官方定义连接 (opens new window)

    语法

    Function.apply(thisArg,[argsArray])

    参数

    thisArg: 可选的。function运行时的this值，但这个this可能会发生变化：在非严格模式下，该参数为null或undefined时会自动替换指向全局对象，原始值会被包装。(同call()函数一致)

    [argsArray]：

    可选的，传递给function的一个参数数组，其中的数组元素将作为单独的参数传个Function函数。如果该参数的值为 null (opens new window) 或 undefined (opens new window)，则表示不需要传入任何参数。

    官方效果：

    function A(name, arg1, arg2) {
    this.name = name
    this.sum = arg1 + arg2
}

function B(name) {
    A.apply(this, [name, 1, 2])
}

let ss = new B("younima")

console.log(ss.name);//younima
console.log(ss.sum);//3

    注：和call()函数比较起来，他们的方法类似，区别在于call()方法接受的是参数列表，而apply接受的是参数数组。

    # 手写实现

    /**
 * @description: 自定义apply实现
 * @param thisArg   上下文this对象
 * @param argArray    对应正常参数数组
 * @return {*}
 */

Function.prototype.newapply = function (thisArg, argArray) {
    thisArg = typeof thisArg === "object" ? thisArg : window
    //保证不覆盖原有属性，实现与function原有属性不关联
    const key = Symbol()
    //this指的是对应的Function方法
    thisArg[key] = this
    //执行函数
    thisArg[key](...argArray)
    //从对象清除该方法，保证对象的纯净
    delete thisArg[key]
}

function A(name, arg1, arg2) {
    this.name = name
    this.sum = arg1 + arg2
}

function B(name) {
    A.newapply(this, [name, 1, 2])
}

let ss = new B("younima")

console.log(ss.name);//younima
console.log(ss.sum);//3

    # 3.bind()实现

    # 定义

    官方定义连接 (opens new window)

    语法

    Function.bind(thisArg,[argsArray])

    参数

    thisArg: 调用绑定函数时作为 this 参数传递给目标函数的值。 如果使用new (opens new window)运算符构造绑定函数，则忽略该值。当使用 bind 在 setTimeout 中创建一个函数（作为回调提供）时，作为 thisArg 传递的任何原始值都将转换为 object。如果 bind 函数的参数列表为空，或者thisArg是null或undefined，执行作用域的 this 将被视为新函数的 thisArg。

    [argsArray]：

    当目标函数被调用时，被预置入绑定函数的参数列表中的参数。

    返回值

    返回一个原函数的拷贝，并拥有指定的this值和初始参数。

    官方效果：

    const A = {
    x: 45,
    getX: function () {
        return this.x
    }
}

const B = A.getX
console.log(B());//undefined

const C = B.bind(A)

console.log(C());//45

    注：bind() 方法创建一个新的函数，在bind()被调用时，这个新函数的this被指定为bind()的第一个参数，而其余参数将作为新函数的参数，供调用时使用。

    # 手写实现

    /**
 * @description: 自定义bind实现
 * @param thisArg   上下文this对象,也是要绑定的this
 * @param *    对应正常参数数组
 * @return {Function}
 */

//  文档中对于bind函数有一个描述，如果是使用new运算符构造的函数，则返回值为新构造的函数，并且this指向这个新函数。
//  所以我们要在这里实现new的功能
// new的具体实现步骤有哪些：
// 1.创建了一个全新的对象。
// 2.将这个对象的原型连接到new的对象的原型上（这个对象会执行[[Prototype]](也就是_proto_)连接）。
// 3.新生成的新对象会绑定到函数调用this。
// 4.通过new生成的对象每个都将被[[Prototype]]连接到这个函数的prototype对象上。
// 5.如果函数没有返回对象类型Object（包含Function，Array，Date，RegExp，Error），那么new表达式中的函数会自动返回这个新的对象。




Function.prototype.newBind = function (thisArg) {
    //保证调用的是一个函数
    if (typeof this !== "function") {
        throw new TypeError(this + "is not a function")
    }
    //存储函数本身
    let self = this
    //取出thisArg的其他参数，转成数组
    const args = [].slice.call(arguments, 1)
    const bound = function () {
        let boundArgs = [].slice.call(arguments)
        //将参数合并为一个新数组
        let finalArgs = args.concat(boundArgs)
        //判断是否使用了new构造符构造新的函数，这里使用this instanceof bound的方法不太严谨，ES6提供了new.target的解决方案
        if (this instanceof bound) {
            //实现上述new方法的1，2，4步
            //1.创建一个全新的对象
            //2.执行[[Prototype]]连接
            //4.通过new生成的对象被[[Prototype]]连接到这个函数的prototype上
            //self函数有可能是箭头函数，并没有prototype，所以没必要加上指向prototype的步骤
            if (self.prototype) {
                function Empty() { }
                Empty.prototype = self.prototype
                bound.prototype = new Empty()
            }
            //这里实现的第三步3.生成的新函数会绑定函数调用的this
            let result = self.apply(this, finalArgs)
            //这里判断第五步
            let isObject = typeof result === "object" || result !== null
            let isFunction = typeof result === "function"
            if (isObject || isFunction) {
                return result
            } else {
                return this
            }
        } else {
            return self.apply(thisArg, finalArgs)

        }
    }
    return bound
}


const A = {
    x: 45,
    getX: function () {
        return this.x
    }
}

const B = A.getX
console.log(B());//undefined

const C = B.newBind(A)

console.log(C());//45
